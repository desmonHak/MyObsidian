DROP TABLE CLIENTE CASCADE CONSTRAINTS;
CREATE TABLE CLIENTE(
    dni       VARCHAR2(9),
    nombre    VARCHAR2(30),
    apellidos VARCHAR2(30),
    fecha_nac DATE,
    CONSTRAINT pk_cliente PRIMARY KEY (dni)
);

DROP TABLE PARCELA CASCADE CONSTRAINTS;
CREATE TABLE PARCELA(
    id                 VARCHAR2(10),
    tamaño             NUMBER,
    ubicacion          VARCHAR2(10),
    CONSTRAINT pk_parcela PRIMARY KEY (id)
);

DROP TABLE RESERVA CASCADE CONSTRAINTS;
CREATE TABLE RESERVA (
    dni              VARCHAR2(9),
    id               VARCHAR2(10),
    f_entrada        DATE,
    f_salida         DATE,
    Constraint pk_reserva PRIMARY KEY (dni, id),
    Constraint fk_reserva1 FOREIGN KEY (dni) REFERENCES CLIENTE(dni),
    Constraint fk_reserva2 FOREIGN KEY (id) REFERENCES PARCELA(id)
);
